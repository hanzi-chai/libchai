32    random_full_key_swap: 0.01,

35impl 变异 for 默认操作 {
36    fn 变异(&mut self, candidate: &mut 元素映射) -> Vec<元素> {
37        let 变异配置 {
38            random_move,
39            random_swap,
40            random_full_key_swap,
41        } = self.变异配置;
42        let sum = random_move + random_swap + random_full_key_swap;
43        let ratio1 = random_move / sum;
44        let ratio2 = (random_move + random_swap) / sum;
45        let number: f64 = random();
46        if number < ratio1 {
47            self.有约束的随机移动(candidate)
48        } else if number < ratio2 {
49            self.有约束的随机交换(candidate)
50        } else {
51            self.有约束的整键随机交换(candidate)
52        }
53    }
54}

56// 默认的问题实现，使用配置文件中的约束来定义各种算子

58    pub fn 新建(数据: &数据) -> Result<Self, 错误> {
59        let (fixed, narrowed) = Self::make_constraints(数据)?;
60        let config = 数据.配置.optimization.clone();
61        let SolverConfig::SimulatedAnnealing(退火方法) = config.unwrap().metaheuristic.unwrap();
62        let 变异配置 = 退火方法.search_method.unwrap_or(DEFAULT_MUTATE);
63        let alphabet: Vec<_> = 数据
64            .配置
65            .form
66            .alphabet
67            .chars()
68            .map(|x| *数据.键转数字.get(&x).unwrap()) // 在生成表示的时候已经确保了这里一定有对应的键
69            .collect();
70        Ok(Self {
71            fixed,
72            narrowed,
73            alphabet,
74            radix: 数据.进制 as usize,
75            elements: 数据.初始映射.len(),
76            变异配置,
77        })
78    }

80    /// 传入配置表示来构造约束，把用户在配置文件中编写的约束「编译」成便于快速计算的数据结构
81    fn make_constraints(
82        representation: &数据,
83    ) -> Result<(HashSet<元素>, HashMap<元素, Vec<键>>), 错误> {
84        let mut fixed: HashSet<元素> = HashSet::new();
85        let mut narrowed: HashMap<元素, Vec<键>> = HashMap::new();
86        let mut values: Vec<AtomicConstraint> = Vec::new();
87        let lookup = |x: String| {
88            let element_number = representation.元素转数字.get(&x);
89            element_number.ok_or(format!("{x} 不存在于键盘映射中"))
90        };
91        let optimization = representation
92            .配置
93            .optimization
94            .as_ref()
95            .ok_or("优化配置不存在")?;
96        if let Some(constraints) = &optimization.constraints {
97            values.append(&mut constraints.elements.clone().unwrap_or_default());
98            values.append(&mut constraints.indices.clone().unwrap_or_default());
99            values.append(&mut constraints.element_indices.clone().unwrap_or_default());
100        }
101        let mapping = &representation.配置.form.mapping;
102        for atomic_constraint in &values {
103            let AtomicConstraint {
104                element,
105                index,
106                keys,
107            } = atomic_constraint;
108            let elements: Vec<usize> = match (element, index) {
109                // 如果指定了元素和码位
110                (Some(element), Some(index)) => {
111                    let element = *lookup(数据::序列化(element, *index))?;
112                    vec![element]
113                }
114                // 如果指定了码位
115                (None, Some(index)) => {
116                    let mut elements = Vec::new();
117                    for (key, value) in mapping {
118                        let normalized = value.normalize();
119                        if let Some(MappedKey::Ascii(_)) = normalized.get(*index) {
120                            let element = *lookup(数据::序列化(key, *index))?;
121                            elements.push(element);
122                        }
123                    }
124                    elements
125                }
126                // 如果指定了元素
127                (Some(element), None) => {
128                    let mapped = mapping
129                        .get(element)
130                        .ok_or(format!("约束中的元素 {element} 不在键盘映射中"))?;
131                    let mut elements = Vec::new();
132                    for (i, x) in mapped.normalize().iter().enumerate() {
133                        if let MappedKey::Ascii(_) = x {
134                            elements.push(*lookup(数据::序列化(element, i))?);
135                        }
136                    }
137                    elements
138                }
139                _ => return Err("约束必须至少提供 element 或 index 之一".into()),
140            };
141            for element in elements {
142                if let Some(keys) = keys {
143                    let mut transformed = Vec::new();
144                    for key in keys {
145                        transformed.push(
146                            *representation
147                                .键转数字
148                                .get(key)
149                                .ok_or(format!("约束中的键 {key} 不在键盘映射中"))?,
150                        );
151                    }
152                    if transformed.is_empty() {
153                        return Err("约束中的键列表不能为空".into());
154                    }
155                    narrowed.insert(element, transformed);
156                } else {
157                    fixed.insert(element);
158                }
159            }
160        }
161        Ok((fixed, narrowed))
162    }

164    fn get_movable_element(&self) -> usize {
165        let mut rng = thread_rng();
166        loop {
167            let key = (self.radix..self.elements).choose(&mut rng).unwrap();
168            if !self.fixed.contains(&key) {
169                return key;
170            }
171        }
172    }

174    fn get_swappable_element(&self) -> usize {
175        let mut rng = thread_rng();
176        loop {
177            let key = (self.radix..self.elements).choose(&mut rng).unwrap();
178            if !self.fixed.contains(&key) {
179                return key;
180            }
181        }
182    }

184    pub fn 有约束的随机交换(&self, keymap: &mut 元素映射) -> Vec<元素> {
185        let element1 = self.get_swappable_element();
186        let key1 = keymap[element1];
187        let mut element2 = self.get_swappable_element();
188        while keymap[element2] == key1 {
189            element2 = self.get_swappable_element();
190        }
191        let key2 = keymap[element2];
192        let destinations1 = self.narrowed.get(&element1).unwrap_or(&self.alphabet);
193        let destinations2 = self.narrowed.get(&element2).unwrap_or(&self.alphabet);
194        //分开判断可行性。这样如果无法交换，至少移动一下。
195        if destinations1.contains(&key2) {
196            keymap[element1] = key2;
197        }
198        if destinations2.contains(&key1) {
199            keymap[element2] = key1;
200        }
201        vec![element1, element2]
202    }

204    pub fn 有约束的整键随机交换(&self, keymap: &mut 元素映射) -> Vec<元素> {
205        let mut rng = thread_rng();
206        // 寻找一个可移动元素和一个它的可行移动位置，然后把这两个键上的所有元素交换
207        // 这样交换不成也至少能移动一次
208        let movable_element = self.get_movable_element();
209        let key1 = keymap[movable_element];
210        let mut destinations = self
211            .narrowed
212            .get(&movable_element)
213            .unwrap_or(&self.alphabet)
214            .clone();
215        destinations.retain(|x| *x != key1);
216        let key2 = destinations.choose(&mut rng).unwrap(); // 在编译约束时已经确保了这里一定有可行的移动位置
217        let mut moved_elements = vec![];
218        for (element, key) in keymap.iter_mut().enumerate() {
219            if *key != key1 && *key != *key2 || self.fixed.contains(&element) {
220                continue;
221            }
222            let destination = if *key == *key2 { key1 } else { *key2 };
223            // 将元素移动到目标
224            let destinations2 = self.narrowed.get(&element).unwrap_or(&self.alphabet);
225            if destinations2.contains(&destination) {
226                *key = destination;
227            }
228            moved_elements.push(element);
229        }
230        moved_elements
231    }

233    pub fn 有约束的随机移动(&self, keymap: &mut 元素映射) -> Vec<元素> {
234        let mut rng = thread_rng();
235        let movable_element = self.get_movable_element();
236        let current = keymap[movable_element];
237        let destinations = self
238            .narrowed
239            .get(&movable_element)
240            .unwrap_or(&self.alphabet);
241        let mut key = destinations.choose(&mut rng).unwrap(); // 在编译约束时已经确保了这里一定有可行的移动位置
242        while *key == current {
243            key = destinations.choose(&mut rng).unwrap();
244        }
245        keymap[movable_element] = *key;
246        vec![movable_element]
247    }
