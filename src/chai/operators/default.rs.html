40// 默认的问题实现，使用配置文件中的约束来定义各种算子
41impl 默认操作 {
42    pub fn 新建(上下文: &默认上下文) -> Result<Self, 错误> {
43        Ok(Self {
44            决策空间: 上下文.决策空间.clone(),
45            元素图: 上下文.元素图.clone(),
46        })
47    }
48
49    fn 传播(&self, 变化: &mut <默认决策 as 决策>::变化, 决策: &mut 默认决策) {
50        // 初始化队列
51        let mut 队列 = VecDeque::new();
52        for 元素 in 变化.iter() {
53            for 下游元素 in self.元素图.get(元素).unwrap_or(&vec![]) {
54                if !队列.contains(下游元素) {
55                    队列.push_back(下游元素.clone());
56                }
57            }
58        }
59        // 传播直到队列为空
60        let mut iters = 0;
61        while !队列.is_empty() {
62            iters += 1;
63            if iters > 100 {
64                panic!("传播超过 100 次仍未结束，可能出现死循环");
65            }
66            let 元素 = 队列.pop_front().unwrap();
67            let mut 合法 = false;
68            let mut 新安排列表 = vec![];
69            for 条件安排 in &self.决策空间.元素[元素] {
70                if 决策.允许(条件安排) {
71                    if 条件安排.安排 == 决策.元素[元素] {
72                        合法 = true;
73                        break;
74                    }
75                    新安排列表.push(条件安排.安排.clone());
76                }
77            }
78            if !合法 {
79                if 新安排列表.is_empty() {
80                    panic!("没有合法的安排，传播失败");
81                } else {
82                    let 新安排 = 新安排列表.choose(&mut rng()).unwrap();
83                    变化.push(元素);
84                    决策.元素[元素] = 新安排.clone();
85                }
86            }
87            for 下游元素 in self.元素图.get(&元素).unwrap_or(&vec![]) {
88                if !队列.contains(下游元素) {
89                    队列.push_back(下游元素.clone());
90                }
91            }
92        }
93    }

95    pub fn 随机移动(&self, 决策: &mut 默认决策) -> Vec<元素> {
96        let mut rng = rng();
97        const MAX_TRIES: usize = 100;
98        for _ in 0..MAX_TRIES {
99            let 元素 = (0..决策.元素.len()).choose(&mut rng).unwrap();
100            // 蓄水池抽样
101            let mut 下一个安排 = None;
102            let mut count = 0;
103            for 条件安排 in &self.决策空间.元素[元素] {
104                if &条件安排.安排 != &决策.元素[元素] && 决策.允许(条件安排) {
105                    count += 1;
106                    if random_range(0..count) == 0 {
107                        下一个安排 = Some(&条件安排.安排);
108                    }
109                }
110            }
111            if let Some(下一个安排) = 下一个安排 {
112                决策.元素[元素] = 下一个安排.clone();
113                return vec![元素];
114            }
115        }
116        vec![]
117    }
