47/// HTTP API: 验证配置
48pub async fn validate_config(Json(config): Json<serde_json::Value>) -> Json<ApiResponse<配置>> {
49    info!("POST /api/validate");
50
51    // 直接在服务器中验证配置
52    match serde_json::from_value::<配置>(config) {
53        Ok(config) => {
54            // 配置解析成功，可以在这里添加额外的验证逻辑
55            // 例如：检查必填字段、验证数值范围等
56            Json(ApiResponse::Success { result: config })
57        }
58        Err(e) => {
59            // 配置解析失败
60            Json(ApiResponse::Error {
61                error: format!("配置解析错误: {}", e),
62            })
63        }
64    }
65}

68pub async fn sync_params(
69    State(state): State<AppState>,
70    Json(params): Json<serde_json::Value>,
71) -> Json<ApiResponse<()>> {
72    info!("POST /api/sync");
73
74    // 直接转换为图形界面参数
75    match serde_json::from_value::<图形界面参数>(params) {
76        Ok(图形界面参数) => {
77            let result = {
78                let mut api = state.api.lock().unwrap();
79                api.sync(图形界面参数)
80            }; // 锁在这里被释放
81
82            match result {
83                Ok(_) => Json(ApiResponse::Success { result: () }),
84                Err(e) => Json(ApiResponse::Error { error: e.message }),
85            }
86        }
87        Err(e) => Json(ApiResponse::Error {
88            error: format!("参数解析错误: {}", e),
89        }),
90    }
91}

121/// HTTP API: 开始优化（异步）
122pub async fn start_optimize(State(state): State<AppState>) -> Json<ApiResponse<String>> {
123    info!("POST /api/optimize");
124
125    // 检查是否已经在运行
126    {
127        let status = state.optimization_status.lock().unwrap();
128        if status.is_running {
129            return Json(ApiResponse::Error {
130                error: "优化已在进行中".to_string(),
131            });
132        }
133    }
134
135    // 设置开始状态
136    {
137        let mut status = state.optimization_status.lock().unwrap();
138        status.is_running = true;
139        status.progress = None;
140        status.error = None;
141    }
142
143    let api = state.api.clone();
144    let status = state.optimization_status.clone();
145
146    // 在后台启动优化任务
147    tokio::spawn(async move {
148        let result = {
149            let api_guard = api.lock().unwrap();
150            api_guard.optimize()
151        }; // 锁在这里被释放
152
153        // 更新最终状态
154        let mut status_guard = status.lock().unwrap();
155        status_guard.is_running = false;
156
157        match result {
158            Ok(_) => {
159                info!("优化完成");
160            }
161            Err(e) => {
162                info!("优化失败: {}", e.message);
163                status_guard.error = Some(e.message);
164            }
165        };
166    });
167
168    Json(ApiResponse::Success {
169        result: "优化已开始，请通过轮询获取进度".to_string(),
170    })
171}

363/// 创建应用路由
364pub fn create_app() -> Router {
365    let state = AppState {
366        api: Arc::new(Mutex::new(WebApi::new())),
367        optimization_status: Arc::new(Mutex::new(OptimizationStatus {
368            is_running: false,
369            progress: None,
370            error: None,
371        })),
372    };
373
374    // 设置全局回调函数
375    {
376        let mut api = state.api.lock().unwrap();
377        let status = state.optimization_status.clone();
378        api.set_callback(move |消息| {
379            // 只记录关键进度
380            match 消息 {
381                crate::消息::Progress { steps, .. } => {
382                    if steps % 100 == 0 {
383                        // 每100步记录一次
384                        info!("优化进度: {} 步", steps);
385                    }
386                }
387                crate::消息::BetterSolution { .. } => {
388                    info!("发现更优解");
389                }
390                crate::消息::Parameters { .. } => {
391                    info!("设置优化参数");
392                }
393                _ => {} // 其他消息不记录，避免日志过多
394            }
395
396            // 更新状态
397            let mut status_guard = status.lock().unwrap();
398            let progress_msg = serde_json::json!(消息);
399            status_guard.progress = Some(progress_msg);
400        });
401    }
402    // 配置更详细的 CORS 设置
403    let cors = CorsLayer::new()
404        .allow_origin(Any)
405        .allow_methods([Method::GET, Method::POST, Method::OPTIONS])
406        .allow_headers(Any)
407        .allow_private_network(true)
408        .allow_credentials(false);
409
410    Router::new()
411        .route("/test", get(index))
412        .route("/api/validate", post(validate_config))
413        .route("/api/sync", post(sync_params))
414        .route("/api/encode", post(encode_evaluate))
415        .route("/api/optimize", post(start_optimize))
416        .route("/api/status", get(get_optimization_status))
417        .fallback_service(ServeDir::new("client"))
418        .layer(DefaultBodyLimit::max(100 * 1024 * 1024)) // 100MB 请求体限制
419        .layer(cors)
420        .layer(TimeoutLayer::new(Duration::from_secs(600))) // 10分钟超时，与编码任务一致
421        .with_state(state)
422}

424/// 尝试绑定可用端口
425async fn bind_available_port(
426    preferred_port: u16,
427) -> Result<(tokio::net::TcpListener, u16), Box<dyn std::error::Error>> {
428    // 首先尝试首选端口
429    let addr = format!("0.0.0.0:{}", preferred_port);
430    match tokio::net::TcpListener::bind(&addr).await {
431        Ok(listener) => {
432            info!("成功绑定到首选端口: {}", preferred_port);
433            return Ok((listener, preferred_port));
434        }
435        Err(e) => {
436            info!("端口 {} 已被占用: {}", preferred_port, e);
437        }
438    }
439
440    // 如果首选端口被占用，尝试附近的端口
441    for offset in 1..=50 {
442        let port = preferred_port + offset;
443        if port < preferred_port {
444            break;
445        }
446
447        let addr = format!("0.0.0.0:{}", port);
448        match tokio::net::TcpListener::bind(&addr).await {
449            Ok(listener) => {
450                info!("成功绑定到替代端口: {}", port);
451                return Ok((listener, port));
452            }
453            Err(_) => {
454                // 继续尝试下一个端口
455            }
456        }
457    }
458
459    // 如果向上寻找失败，尝试向下寻找
460    for offset in 1..=50 {
461        if preferred_port < offset {
462            break;
463        }
464
465        let port = preferred_port - offset;
466        if port < 1024 {
467            // 避免使用系统保留端口
468            break;
469        }
470
471        let addr = format!("0.0.0.0:{}", port);
472        match tokio::net::TcpListener::bind(&addr).await {
473            Ok(listener) => {
474                info!("成功绑定到替代端口: {}", port);
475                return Ok((listener, port));
476            }
477            Err(_) => {
478                // 继续尝试下一个端口
479            }
480        }
481    }
482
483    // 最后尝试让系统自动分配端口
484    match tokio::net::TcpListener::bind("0.0.0.0:0").await {
485        Ok(listener) => {
486            let actual_port = listener.local_addr()?.port();
487            info!("使用系统自动分配的端口: {}", actual_port);
488            Ok((listener, actual_port))
489        }
490        Err(e) => Err(format!("无法绑定到任何端口: {}", e).into()),
491    }
492}

494/// 启动服务器
495pub async fn start_server(port: u16) -> Result<(), Box<dyn std::error::Error>> {
496    tracing_subscriber::fmt::init();
497
498    let app = create_app();
499
500    // 尝试绑定端口，如果失败则尝试其他端口
501    let (listener, actual_port) = bind_available_port(port).await?;
502
503    info!("Listening on: http://127.0.0.1:{}", actual_port);
504    info!("API Endpoints:");
505    info!("   POST /api/validate    - 验证配置");
506    info!("   POST /api/sync        - 同步参数");
507    info!("   POST /api/encode      - 编码评估");
508    info!("   POST /api/optimize    - 开始优化");
509    info!("   GET  /api/status      - 获取优化状态");
510
511    axum::serve(listener, app).await?;
512
513    Ok(())
