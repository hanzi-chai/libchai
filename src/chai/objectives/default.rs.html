48impl<E: 编码器<解类型 = 元素映射>> 默认目标函数<E> {
49    /// 通过传入配置表示、编码器和共用资源来构造一个目标函数
50    pub fn 新建(上下文: &默认上下文, 编码器: E) -> Result<Self, 错误> {
51        let 键位分布信息 = 上下文.键位分布信息.clone();
52        let 当量信息 = 上下文.当量信息.clone();
53        let 正则化 = if let Some(正则化配置) = 上下文
54            .配置
55            .optimization
56            .clone()
57            .map(|x| x.objective)
58            .and_then(|x| x.regularization)
59        {
60            默认上下文::预处理正则化(&正则化配置, &上下文.棱镜.元素转数字)?
61        } else {
62            FxHashMap::default()
63        };
64        let 指法计数 = 上下文.棱镜.预处理指法标记(上下文.get_space());
65        let config = 上下文
66            .配置
67            .optimization
68            .as_ref()
69            .ok_or("优化配置不存在")?
70            .objective
71            .clone();
72        let 最大编码 = 当量信息.len() as u64;
73        let 构造缓存 =
74            |x: &PartialWeights| 缓存::new(x, 上下文.棱镜.进制, 上下文.词列表.len(), 最大编码);
75        let 一字全码 = config.characters_full.as_ref().map(构造缓存);
76        let 一字简码 = config.characters_short.as_ref().map(构造缓存);
77        let 多字全码 = config.words_full.as_ref().map(构造缓存);
78        let 多字简码 = config.words_short.as_ref().map(构造缓存);
79        let 计数桶列表 = vec![[一字全码, 一字简码], [多字全码, 多字简码]];
80        let 参数 = 默认目标函数参数 {
81            键位分布信息,
82            当量信息,
83            指法计数,
84            数字转键: 上下文.棱镜.数字转键.clone(),
85            正则化,
86            正则化强度: config
87                .regularization
88                .and_then(|x| x.strength)
89                .unwrap_or(1.0),
90        };
91        let 编码结果: Vec<_> = 上下文.词列表.iter().map(编码信息::new).collect();
92        Ok(Self {
93            参数,
94            编码器,
95            编码结果: 编码结果.clone(),
96            计数桶列表: 计数桶列表.clone(),
97        })
98    }
99}

101impl<E: 编码器<解类型 = 元素映射>> 目标函数 for 默认目标函数<E> {
102    type 目标值 = 默认指标;
103    type 解类型 = 元素映射;
104
105    /// 计算各个部分编码的指标，然后将它们合并成一个指标输出
106    fn 计算(&mut self, 解: &元素映射, 变化: &Option<Vec<元素>>) -> (默认指标, f64) {
107        let 参数 = &self.参数;
108        self.编码器.编码(解, 变化, &mut self.编码结果);
109        let mut 桶序号列表: Vec<_> = self.计数桶列表.iter().map(|_| 0).collect();
110        // 开始计算指标
111        for 编码信息 in self.编码结果.iter_mut() {
112            let 频率 = 编码信息.频率;
113            let 桶索引 = if 编码信息.词长 == 1 { 0 } else { 1 };
114            let 桶 = &mut self.计数桶列表[桶索引];
115            let 桶序号 = 桶序号列表[桶索引];
116            if let Some(缓存) = &mut 桶[0] {
117                缓存.处理(桶序号, 频率, &mut 编码信息.全码, 参数);
118            }
119            if let Some(缓存) = &mut 桶[1] {
120                缓存.处理(桶序号, 频率, &mut 编码信息.简码, 参数);
121            }
122            桶序号列表[桶索引] += 1;
123        }
124
125        let mut 目标函数 = 0.0;
126        let mut 指标 = 默认指标 {
127            characters_full: None,
128            words_full: None,
129            characters_short: None,
130            words_short: None,
131            memory: None,
132        };
133        for (桶索引, 桶) in self.计数桶列表.iter().enumerate() {
134            let _ = &桶[0].as_ref().map(|x| {
135                let (分组指标, 分组目标函数) = x.汇总(参数);
136                目标函数 += 分组目标函数;
137                if 桶索引 == 0 {
138                    指标.characters_full = Some(分组指标);
139                } else {
140                    指标.words_full = Some(分组指标);
141                }
142            });
143            let _ = &桶[1].as_ref().map(|x| {
144                let (分组指标, 分组目标函数) = x.汇总(参数);
145                目标函数 += 分组目标函数;
146                if 桶索引 == 0 {
147                    指标.characters_short = Some(分组指标);
148                } else {
149                    指标.words_short = Some(分组指标);
150                }
151            });
152        }
153
154        if !参数.正则化.is_empty() {
155            let mut 记忆量 = 解.len() as f64;
156            for (元素, 键) in 解.iter().enumerate() {
157                if 元素 as u64 == *键 {
158                    记忆量 -= 1.0;
159                    continue;
160                }
161                if let Some(归并列表) = 参数.正则化.get(&元素) {
162                    let mut 最大亲和度 = 0.0;
163                    for (目标元素, 亲和度) in 归并列表.iter() {
164                        if 解[*目标元素] == *键 {
165                            最大亲和度 = 亲和度.max(最大亲和度);
166                        }
167                    }
168                    记忆量 -= 最大亲和度;
169                }
170            }
171            指标.memory = Some(记忆量);
172            let 归一化记忆量 = 记忆量 / 解.len() as f64;
173            目标函数 += 归一化记忆量 * 参数.正则化强度;
174        }
175        (指标, 目标函数)
176    }
