49/// 目标函数
50impl<E: 编码器<解类型 = 元素映射>> 默认目标函数<E> {
51    /// 通过传入配置表示、编码器和共用资源来构造一个目标函数
52    pub fn 新建(上下文: &默认上下文, 编码器: E) -> Result<Self, 错误> {
53        let 键位分布信息 = 上下文.键位分布信息.clone();
54        let 当量信息 = 上下文.当量信息.clone();
55        let 正则化 = if let Some(正则化配置) = 上下文
56            .配置
57            .optimization
58            .clone()
59            .map(|x| x.objective)
60            .and_then(|x| x.regularization)
61        {
62            默认上下文::预处理正则化(&正则化配置, &上下文.棱镜.元素转数字)?
63        } else {
64            FxHashMap::default()
65        };
66        let 指法计数 = 上下文.棱镜.预处理指法标记(上下文.get_space());
67        let config = 上下文
68            .配置
69            .optimization
70            .as_ref()
71            .ok_or("优化配置不存在")?
72            .objective
73            .clone();
74        let 最大编码 = 当量信息.len() as u64;
75        let 构造缓存 =
76            |x: &PartialWeights| 缓存::new(x, 上下文.棱镜.进制, 上下文.词列表.len(), 最大编码);
77        let 一字全码 = config.characters_full.as_ref().map(构造缓存);
78        let 一字简码 = config.characters_short.as_ref().map(构造缓存);
79        let 多字全码 = config.words_full.as_ref().map(构造缓存);
80        let 多字简码 = config.words_short.as_ref().map(构造缓存);
81        let 计数桶列表 = vec![[一字全码, 一字简码], [多字全码, 多字简码]];
82        let 参数 = 默认目标函数参数 {
83            键位分布信息,
84            当量信息,
85            指法计数,
86            数字转键: 上下文.棱镜.数字转键.clone(),
87            正则化,
88            正则化强度: config
89                .regularization
90                .and_then(|x| x.strength)
91                .unwrap_or(1.0),
92        };
93        let 编码结果: Vec<_> = 上下文.词列表.iter().map(编码信息::new).collect();
94        Ok(Self {
95            参数,
96            编码器,
97            编码结果: 编码结果.clone(),
98            计数桶列表: 计数桶列表.clone(),
99            编码结果缓冲: 编码结果.clone(),
100            计数桶列表缓冲: 计数桶列表.clone(),
101        })
102    }
103}

105impl<E: 编码器<解类型 = 元素映射>> 目标函数 for 默认目标函数<E> {
106    type 目标值 = 默认指标;
107    type 解类型 = 元素映射;
108
109    /// 计算各个部分编码的指标，然后将它们合并成一个指标输出
110    fn 计算(&mut self, 解: &元素映射, 变化: &Option<Vec<元素>>) -> (默认指标, f64) {
111        let 参数 = &self.参数;
112        self.编码结果缓冲.clone_from(&self.编码结果);
113        self.编码器.编码(解, 变化, &mut self.编码结果缓冲);
114        self.计数桶列表缓冲.clone_from(&self.计数桶列表);
115        let mut 桶序号列表: Vec<_> = self.计数桶列表缓冲.iter().map(|_| 0).collect();
116        // 开始计算指标
117        for 编码信息 in self.编码结果缓冲.iter_mut() {
118            let 频率 = 编码信息.频率;
119            let 桶索引 = if 编码信息.词长 == 1 { 0 } else { 1 };
120            let 桶 = &mut self.计数桶列表缓冲[桶索引];
121            let 桶序号 = 桶序号列表[桶索引];
122            if let Some(缓存) = &mut 桶[0] {
123                缓存.处理(桶序号, 频率, &mut 编码信息.全码, 参数);
124            }
125            if let Some(缓存) = &mut 桶[1] {
126                缓存.处理(桶序号, 频率, &mut 编码信息.简码, 参数);
127            }
128            桶序号列表[桶索引] += 1;
129        }
130
131        if 变化.is_none() {
132            self.编码结果.clone_from(&self.编码结果缓冲);
133            self.计数桶列表.clone_from(&self.计数桶列表缓冲);
134        }
135
136        let mut 目标函数 = 0.0;
137        let mut 指标 = 默认指标 {
138            characters_full: None,
139            words_full: None,
140            characters_short: None,
141            words_short: None,
142            memory: None,
143        };
144        for (桶索引, 桶) in self.计数桶列表缓冲.iter().enumerate() {
145            let _ = &桶[0].as_ref().map(|x| {
146                let (分组指标, 分组目标函数) = x.汇总(参数);
147                目标函数 += 分组目标函数;
148                if 桶索引 == 0 {
149                    指标.characters_full = Some(分组指标);
150                } else {
151                    指标.words_full = Some(分组指标);
152                }
153            });
154            let _ = &桶[1].as_ref().map(|x| {
155                let (分组指标, 分组目标函数) = x.汇总(参数);
156                目标函数 += 分组目标函数;
157                if 桶索引 == 0 {
158                    指标.characters_short = Some(分组指标);
159                } else {
160                    指标.words_short = Some(分组指标);
161                }
162            });
163        }
164
165        if !参数.正则化.is_empty() {
166            let mut 记忆量 = 解.len() as f64;
167            for (元素, 键) in 解.iter().enumerate() {
168                if 元素 as u64 == *键 {
169                    记忆量 -= 1.0;
170                    continue;
171                }
172                if let Some(归并列表) = 参数.正则化.get(&元素) {
173                    let mut 最大亲和度 = 0.0;
174                    for (目标元素, 亲和度) in 归并列表.iter() {
175                        if 解[*目标元素] == *键 {
176                            最大亲和度 = 亲和度.max(最大亲和度);
177                        }
178                    }
179                    记忆量 -= 最大亲和度;
180                }
181            }
182            指标.memory = Some(记忆量);
183            let 归一化记忆量 = 记忆量 / 解.len() as f64;
184            目标函数 += 归一化记忆量 * 参数.正则化强度;
185        }
186        (指标, 目标函数)
187    }
188
189    fn 接受新解(&mut self) {
190        self.编码结果.clone_from(&self.编码结果缓冲);
191        self.计数桶列表.clone_from(&self.计数桶列表缓冲);
192    }
193}
