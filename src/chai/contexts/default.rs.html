28impl 上下文 for 默认上下文 {
29    type 解类型 = 元素映射;
30    fn 序列化(&self, 解: &Self::解类型) -> String {
31        let mut new_config = self.配置.clone();
32        let lookup = |element: &String| {
33            let number = *self.棱镜.元素转数字.get(element).unwrap(); // 输入的时候已经检查过一遍，不需要再次检查
34            let current_mapped = &解[number];
35            *self.棱镜.数字转键.get(current_mapped).unwrap() // 同上
36        };
37        for (element, mapped) in &self.配置.form.mapping {
38            let new_element = element.clone();
39            let new_mapped = match mapped {
40                Mapped::Basic(string) => {
41                    let mut all_codes = String::new();
42                    for index in 0..string.len() {
43                        let name = 元素标准名称(element, index);
44                        all_codes.push(lookup(&name));
45                    }
46                    Mapped::Basic(all_codes)
47                }
48                Mapped::Advanced(vector) => {
49                    let all_codes: Vec<MappedKey> = vector
50                        .iter()
51                        .enumerate()
52                        .map(|(index, mapped_key)| match mapped_key {
53                            MappedKey::Ascii(_) => {
54                                MappedKey::Ascii(lookup(&元素标准名称(element, index)))
55                            }
56                            other => other.clone(),
57                        })
58                        .collect();
59                    Mapped::Advanced(all_codes)
60                }
61                x => x.clone()
62            };
63            new_config.form.mapping.insert(new_element, new_mapped);
64        }
65        to_string(&new_config).unwrap()
66    }
67}

70    pub fn 新建(输入: 默认输入) -> Result<Self, 错误> {
71        let (初始映射, 选择键, 棱镜) = Self::构建棱镜(&输入.配置)?;
72        let 最大码长 = 输入.配置.encoder.max_length;
73        let 词列表 = 棱镜.预处理词列表(输入.词列表, 最大码长)?;
74        let 组合长度 = 最大码长.min(最大按键组合长度);
75        let 编码空间大小 = 棱镜.进制.pow(组合长度 as u32) as usize;
76        let 键位分布信息 = 棱镜.预处理键位分布信息(&输入.原始键位分布信息);
77        let 当量信息 = 棱镜.预处理当量信息(&输入.原始当量信息, 编码空间大小);
78        Ok(Self {
79            配置: 输入.配置,
80            词列表,
81            键位分布信息,
82            当量信息,
83            初始映射,
84            棱镜,
85            选择键,
86        })
87    }

89    /// 读取字母表和选择键列表，然后分别对它们的每一个按键转换成无符号整数
90    /// 1, ... n = 所有常规编码键
91    /// n + 1, ..., m = 所有选择键
92    pub fn 构建棱镜(配置: &配置) -> Result<(元素映射, Vec<键>, 棱镜), 错误> {
93        let mut 键转数字: FxHashMap<char, 键> = FxHashMap::default();
94        let mut 数字转键: FxHashMap<键, char> = FxHashMap::default();
95        let mut 数字 = 1;
96        for 键 in 配置.form.alphabet.chars() {
97            if 键转数字.contains_key(&键) {
98                return Err("编码键有重复！".into());
99            };
100            键转数字.insert(键, 数字);
101            数字转键.insert(数字, 键);
102            数字 += 1;
103        }
104        let 默认选择键 = vec!['_'];
105        let 原始选择键 = 配置.encoder.select_keys.as_ref().unwrap_or(&默认选择键);
106        if 原始选择键.is_empty() {
107            return Err("选择键不能为空！".into());
108        }
109        let mut 选择键: Vec<键> = vec![];
110        for 键 in 原始选择键 {
111            if 键转数字.contains_key(键) {
112                return Err("编码键有重复！".into());
113            };
114            键转数字.insert(*键, 数字);
115            数字转键.insert(数字, *键);
116            选择键.push(数字);
117            数字 += 1;
118        }
119        let 进制 = 数字;
120        let mut 元素映射: 元素映射 = (0..进制).collect();
121        let mut 元素转数字: FxHashMap<String, 元素> = FxHashMap::default();
122        let mut 数字转元素: FxHashMap<元素, String> = FxHashMap::default();
123        for (键字符, 键) in &键转数字 {
124            元素转数字.insert(键字符.to_string(), *键 as usize);
125            数字转元素.insert(*键 as usize, 键字符.to_string());
126        }
127        for (元素, 映射值) in &配置.form.mapping {
128            let 映射值 = 映射值.normalize();
129            for (序号, 映射键) in 映射值.iter().enumerate() {
130                if let MappedKey::Ascii(x) = 映射键 {
131                    if let Some(键) = 键转数字.get(x) {
132                        let 元素名 = 元素标准名称(元素, 序号);
133                        元素转数字.insert(元素名.clone(), 元素映射.len());
134                        数字转元素.insert(元素映射.len(), 元素名.clone());
135                        元素映射.push(*键);
136                    } else {
137                        return Err(format!("元素 {元素} 的编码中的字符 {x} 并不在字母表中").into());
138                    }
139                }
140            }
141        }
142        let 棱镜 = 棱镜 {
143            键转数字,
144            数字转键,
145            元素转数字,
146            数字转元素,
147            进制,
148        };
149        Ok((元素映射, 选择键, 棱镜))
150    }

152    pub fn 预处理正则化(
153        正则化: &Regularization,
154        元素转数字: &FxHashMap<String, 元素>,
155    ) -> Result<FxHashMap<元素, Vec<(元素, f64)>>, 错误> {
156        let mut result = FxHashMap::default();
157        if let Some(列表) = &正则化.element_affinities {
158            for 规则 in 列表 {
159                let 元素名称 = 元素标准名称(&规则.from.element, 规则.from.index);
160                let 元素 = 元素转数字
161                    .get(&元素名称)
162                    .ok_or(format!("元素 {元素名称} 不存在"))?;
163                let mut 亲和度列表 = Vec::new();
164                for 目标 in 规则.to.iter() {
165                    let 目标元素名称 =
166                        元素标准名称(&目标.element.element, 目标.element.index);
167                    let 目标元素 = 元素转数字
168                        .get(&目标元素名称)
169                        .ok_or(format!("目标元素 {目标元素名称} 不存在"))?;
170                    亲和度列表.push((*目标元素, 目标.affinity));
171                }
172                result.insert(*元素, 亲和度列表);
173            }
174        }
175        if let Some(列表) = &正则化.key_affinities {
176            for 规则 in 列表 {
177                let 元素名称 = 元素标准名称(&规则.from.element, 规则.from.index);
178                let 元素 = 元素转数字
179                    .get(&元素名称)
180                    .ok_or(format!("元素 {元素名称} 不存在"))?;
181                let mut 亲和度列表 = Vec::new();
182                for 目标 in 规则.to.iter() {
183                    let 目标键位 = 元素转数字
184                        .get(&目标.key.to_string())
185                        .ok_or("目标键位不存在")?;
186                    亲和度列表.push((*目标键位, 目标.affinity));
187                }
188                result.insert(*元素, 亲和度列表);
189            }
190        }
191        Ok(result)
192    }

194    pub fn 生成码表(&self, 编码结果: &[编码信息]) -> Vec<码表项> {
195        let mut 码表: Vec<(usize, 码表项)> = Vec::new();
196        let 转编码 = |code: 编码| self.棱镜.数字转编码(code).iter().collect();
197        for (序号, 可编码对象) in self.词列表.iter().enumerate() {
198            let 码表项 = 码表项 {
199                name: 可编码对象.名称.clone(),
200                full: 转编码(编码结果[序号].全码.原始编码),
201                full_rank: 编码结果[序号].全码.原始编码候选位置,
202                short: 转编码(编码结果[序号].简码.原始编码),
203                short_rank: 编码结果[序号].简码.原始编码候选位置,
204            };
205            码表.push((可编码对象.原始顺序, 码表项));
206        }
207        码表.sort_by_key(|x| x.0);
208        码表.into_iter().map(|x| x.1).collect()
209    }

211    /// 将编码空间内所有的编码组合预先计算好是否能自动上屏
212    /// 按照这个字符串所对应的整数为下标，存储到一个大数组中
213    pub fn 预处理自动上屏(&self) -> Result<Vec<bool>, 错误> {
214        let mut result: Vec<bool> = vec![];
215        let encoder = &self.配置.encoder;
216        let mut re: Option<Regex> = None;
217        if let Some(pattern) = &encoder.auto_select_pattern {
218            let re_or_error = Regex::new(pattern);
219            if let Ok(regex) = re_or_error {
220                re = Some(regex);
221            } else {
222                return Err(format!("正则表达式 {pattern} 无法解析").into());
223            }
224        }
225        for code in 0..self.get_space() {
226            let chars = self.棱镜.数字转编码(code as u64);
227            let string: String = chars.iter().collect();
228            let is_matched = if let Some(re) = &re {
229                re.is_match(&string)
230            } else if let Some(length) = encoder.auto_select_length {
231                chars.len() >= length
232            } else {
233                true
234            };
235            let is_max_length = chars.len() == encoder.max_length;
236            result.push(is_matched || is_max_length);
237        }
238        Ok(result)
239    }

241    pub fn 预处理简码规则(
242        &self,
243        schemes: &Vec<Scheme>,
244    ) -> Result<Vec<简码配置>, 错误> {
245        let mut compiled_schemes = Vec::new();
246        for scheme in schemes {
247            let prefix = scheme.prefix;
248            let count = scheme.count.unwrap_or(1);
249            let select_keys = if let Some(keys) = &scheme.select_keys {
250                let mut transformed_keys = Vec::new();
251                for key in keys {
252                    let transformed_key = self
253                        .棱镜
254                        .键转数字
255                        .get(key)
256                        .ok_or(format!("简码的选择键 {key} 不在全局选择键中"))?;
257                    transformed_keys.push(*transformed_key);
258                }
259                transformed_keys
260            } else {
261                self.选择键.clone()
262            };
263            if count > select_keys.len() {
264                return Err("选重数量不能高于选择键数量".into());
265            }
266            compiled_schemes.push(简码配置 {
267                prefix,
268                select_keys: select_keys[..count].to_vec(),
269            });
270        }
271        Ok(compiled_schemes)
272    }

274    pub fn 预处理简码配置(
275        &self,
276        原始简码配置列表: Vec<ShortCodeConfig>,
277    ) -> Result<[Vec<简码配置>; 最大词长], 错误> {
278        let mut short_code: [Vec<简码配置>; 最大词长] = Default::default();
279        for config in 原始简码配置列表 {
280            match config {
281                ShortCodeConfig::Equal {
282                    length_equal,
283                    schemes,
284                } => {
285                    short_code[length_equal - 1].extend(self.预处理简码规则(&schemes)?);
286                }
287                ShortCodeConfig::Range {
288                    length_in_range: (from, to),
289                    schemes,
290                } => {
291                    for length in from..=to {
292                        short_code[length - 1].extend(self.预处理简码规则(&schemes)?);
293                    }
294                }
295            }
296        }
297        Ok(short_code)
298    }

300    pub fn get_space(&self) -> usize {
301        let max_length = self.配置.encoder.max_length.min(最大按键组合长度);
302        self.棱镜.进制.pow(max_length as u32) as usize
303    }
