14/// 通过图形界面来使用 libchai 的入口，实现了界面特征
15#[wasm_bindgen]

21/// 用于在图形界面验证输入的配置是否正确
22#[wasm_bindgen]
23pub fn validate(js_config: JsValue) -> Result<JsValue, JsError> {
24    set_once();
25    let 配置: 配置 = from_value(js_config)?;
26    let 序列化 = Serializer::json_compatible();
27    Ok(配置.serialize(&序列化)?)
28}

30#[wasm_bindgen]
31impl Web {
32    pub fn new(回调: Function) -> Web {
33        set_once();
34        let 参数 = 默认输入::default();
35        Self { 回调, 参数 }
36    }
37
38    pub fn sync(&mut self, 前端参数: JsValue) -> Result<(), JsError> {
39        self.参数 = from_value(前端参数)?;
40        Ok(())
41    }
42
43    pub fn encode_evaluate(&self, 前端目标函数配置: JsValue) -> Result<JsValue, JsError> {
44        let 目标函数配置: ObjectiveConfig = from_value(前端目标函数配置)?;
45        let mut 输入 = self.参数.clone();
46        输入.配置.optimization = Some(OptimizationConfig {
47            objective: 目标函数配置,
48            constraints: None,
49            metaheuristic: None,
50        });
51        let 上下文 = 默认上下文::新建(输入)?;
52        let 编码器 = 默认编码器::新建(&上下文)?;
53        let mut 目标函数 = 默认目标函数::新建(&上下文, 编码器)?;
54        let (指标, _) = 目标函数.计算(&上下文.初始映射, &None);
55        let 码表 = 上下文.生成码表(&目标函数.编码结果);
56        Ok(to_value(&(码表, 指标))?)
57    }
58
59    pub fn optimize(&self) -> Result<(), JsError> {
60        let 优化方法配置 = self.参数.配置.clone().optimization.unwrap().metaheuristic.unwrap();
61        let 上下文 = 默认上下文::新建(self.参数.clone())?;
62        let 编码器 = 默认编码器::新建(&上下文)?;
63        let mut 目标函数 = 默认目标函数::新建(&上下文, 编码器)?;
64        let mut 操作 = 默认操作::新建(&上下文)?;
65        let SolverConfig::SimulatedAnnealing(退火) = 优化方法配置;
66        退火.优化(&上下文.初始映射, &mut 目标函数, &mut 操作, &上下文, self);
67        Ok(())
68    }
69}

71impl 界面 for Web {
72    fn 发送(&self, 消息: 消息) {
73        let 序列化 = Serializer::json_compatible();
74        let 前端消息 = 消息.serialize(&序列化).unwrap();
75        self.回调.call1(&JsValue::null(), &前端消息).unwrap();
76    }
