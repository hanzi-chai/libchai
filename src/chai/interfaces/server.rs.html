2use crate::{

12/// 纯 Rust 的 Web API 接口，与 wasm_bindgen Web 结构一一对应
13#[derive(Default)]
14pub struct WebApi {
15    参数: 默认输入,
16    回调: Option<Box<dyn Fn(&消息) + Send + Sync>>,
17}
18
19impl WebApi {
20    /// 创建新的 WebApi 实例，与 Web::new 对应
21    pub fn new() -> Self {
22        set_once();
23        let 参数 = 默认输入::default();
24        Self {
25            参数, 回调: None
26        }
27    }
28
29    /// 设置消息回调函数
30    pub fn set_callback<F>(&mut self, callback: F)
31    where
32        F: Fn(&消息) + Send + Sync + 'static,
33    {
34        self.回调 = Some(Box::new(callback));
35    }
36
37    /// 同步前端参数，与 Web::sync 对应
38    pub fn sync(&mut self, 前端参数: 默认输入) -> Result<(), 错误> {
39        self.参数 = 前端参数;
40        Ok(())
41    }
42
43    /// 编码评估，与 Web::encode_evaluate 对应
44    pub fn encode_evaluate(
45        &self,
46        目标函数配置: ObjectiveConfig,
47    ) -> Result<(Vec<码表项>, 默认指标), 错误> {
48        let mut 输入 = self.参数.clone();
49        输入.配置.optimization = Some(OptimizationConfig {
50            objective: 目标函数配置,
51            constraints: None,
52            metaheuristic: None,
53        });
54        let 上下文 = 默认上下文::新建(输入)?;
55        let 编码器 = 默认编码器::新建(&上下文)?;
56        let mut 目标函数 = 默认目标函数::新建(&上下文, 编码器)?;
57        let (指标, _) = 目标函数.计算(&上下文.初始映射, &None);
58        let 码表 = 上下文.生成码表(&目标函数.编码结果);
59
60        Ok((码表, 指标))
61    }
62
63    /// 优化，与 Web::optimize 对应  
64    pub fn optimize(&self) -> Result<(), 错误> {
65        let 优化方法配置 = self.参数.配置.clone().optimization.unwrap().metaheuristic.unwrap();
66        let 上下文 = 默认上下文::新建(self.参数.clone())?;
67        let 编码器 = 默认编码器::新建(&上下文)?;
68        let mut 目标函数 = 默认目标函数::新建(&上下文, 编码器)?;
69        let mut 操作 = 默认操作::新建(&上下文)?;
70        let SolverConfig::SimulatedAnnealing(退火) = 优化方法配置;
71        退火.优化(&上下文.初始映射, &mut 目标函数, &mut 操作, &上下文, self);
72        Ok(())
73    }
74}
75
76impl 界面 for WebApi {
77    fn 发送(&self, 消息: 消息) {
78        if let Some(ref callback) = self.回调 {
79            callback(&消息);
80        }
81    }
