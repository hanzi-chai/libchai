16#[derive(Debug, Clone, Copy, Serialize, Deserialize)]
17/// 退火算法的参数，包括最高温、最低温、步数
18pub struct 降温时间表 {
19    pub t_max: f64,
20    pub t_min: f64,
21    pub steps: usize,
22}
23
24#[skip_serializing_none]
25#[derive(Debug, Clone, Serialize, Deserialize)]
26pub struct 退火方法 {
27    pub parameters: Option<降温时间表>,
28    pub report_after: Option<f64>,
29    pub search_method: Option<变异配置>,
30    pub update_interval: Option<usize>,
31}
32
33impl<F: 变异> 优化方法<F> for 退火方法 {
34    fn 优化<E: 编码器, O: 目标函数>(
35        &self,
36        问题: &mut 优化问题<E, O, F>,
37        界面: &dyn 界面,
38    ) -> 优化结果<O> {
39        let 降温时间表 = self.parameters.unwrap_or_else(|| self.调参(问题, 界面));
40        self.solve_with(问题, 降温时间表, 界面)
41    }
42}
43
44impl 退火方法 {
45    /// 退火算法求解的主函数
46    fn solve_with<E: 编码器, O: 目标函数, F: 变异>(
47        &self,
48        问题: &mut 优化问题<E, O, F>,
49        parameters: 降温时间表,
50        interface: &dyn 界面,
51    ) -> 优化结果<O> {
52        let mut best_candidate = 问题.数据.初始映射.clone();
53        let mut best_rank = 问题.计算(&best_candidate, &None);
54        let mut annealing_candidate = best_candidate.clone();
55        let mut annealing_rank = best_rank.clone();
56        let mut last_diff = vec![];
57        let 降温时间表 {
58            t_max,
59            t_min,
60            steps,
61        } = parameters;
62        let start = Instant::now();
63        let update_interval = self.update_interval.unwrap_or(1000);
64
65        for step in 0..steps {
66            // 等比级数降温：每一步的温度都是上一步的温度乘以一个固定倍数
67            let progress = step as f64 / steps as f64;
68            let temperature = t_max * (t_min / t_max).powf(progress);
69            // 每过一定的步数，报告当前状态和计算速度
70            if step % update_interval == 0 || step == steps - 1 {
71                interface.发送(消息::Progress {
72                    steps: step,
73                    temperature,
74                    metric: format!("{}", annealing_rank.0),
75                });
76                if step == update_interval {
77                    let elapsed = start.elapsed().as_micros() as u64 / update_interval as u64;
78                    interface.发送(消息::Elapsed { time: elapsed });
79                }
80            }
81            // 生成一个新解
82            let mut next_candidate = annealing_candidate.clone();
83            let diff = 问题.操作.变异(&mut next_candidate);
84            let mut total_diff = diff.clone();
85            total_diff.extend(&last_diff);
86            let next_rank = 问题.计算(&next_candidate, &Some(total_diff));
87            // 如果满足退火条件，接受新解
88            let improvement = next_rank.1 - annealing_rank.1;
89            if improvement < 0.0 || (random::<f64>() < (-improvement / temperature).exp()) {
90                annealing_candidate.clone_from(&next_candidate);
91                annealing_rank = next_rank;
92                last_diff.clear();
93            } else {
94                last_diff = diff;
95            }
96            // 如果当前解优于目前的最优解，更新最优解
97            if annealing_rank.1 < best_rank.1 {
98                best_rank = annealing_rank.clone();
99                best_candidate.clone_from(&annealing_candidate);
100                let save = progress > self.report_after.unwrap_or(0.9);
101                interface.发送(消息::BetterSolution {
102                    metric: format!("{}", best_rank.0),
103                    config: 问题.数据.更新配置(&best_candidate),
104                    save,
105                })
106            }
107        }
108        interface.发送(消息::BetterSolution {
109            metric: format!("{}", best_rank.0),
110            config: 问题.数据.更新配置(&best_candidate),
111            save: true,
112        });
113        优化结果 {
114            映射: best_candidate,
115            指标: best_rank.0.clone(),
116            分数: best_rank.1,
117        }
118    }
119
120    fn trial_run<E: 编码器, O: 目标函数, F: 变异>(
121        &self,
122        问题: &mut 优化问题<E, O, F>,
123        from: 元素映射,
124        temperature: f64,
125        steps: usize,
126    ) -> (元素映射, f64, f64) {
127        let mut candidate = from.clone();
128        let (_, mut energy) = 问题.计算(&candidate, &None);
129        let mut accepts = 0;
130        let mut improves = 0;
131
132        for _ in 0..steps {
133            let mut next_candidate = candidate.clone();
134            let moved_elements = 问题.操作.变异(&mut next_candidate);
135            let (_, next_energy) = 问题.计算(&next_candidate, &Some(moved_elements));
136            let energy_delta = next_energy - energy;
137            if energy_delta < 0.0 || (-energy_delta / temperature).exp() > random::<f64>() {
138                accepts += 1;
139                if energy_delta < 0.0 {
140                    improves += 1;
141                }
142                candidate = next_candidate;
143                energy = next_energy;
144            }
145        }
146        let accept_rate = accepts as f64 / steps as f64;
147        let improve_rate = improves as f64 / steps as f64;
148        (candidate, accept_rate, improve_rate)
149    }

151    // 不提供参数，通过试验来获得一组参数的办法
152    pub fn 调参<E: 编码器, O: 目标函数, F: 变异>(
153        &self,
154        问题: &mut 优化问题<E, O, F>,
155        界面: &dyn 界面,
156    ) -> 降温时间表 {
157        // 最高温时，接受概率应该至少有这么多
158        const HIGH_ACCEPTANCE: f64 = 0.98;
159        // 最低温时，改进概率应该至多有这么多
160        const LOW_IMPROVEMENT: f64 = 0.02;
161        // 搜索温度时用的步进大小
162        const MULTIPLIER: f64 = 2.0;
163
164        let batch = 1000;
165        let mut candidate = 问题.数据.初始映射.clone();
166        let (_, energy) = 问题.计算(&candidate, &None);
167        let mut sum_delta = 0.0;
168        for _ in 0..batch {
169            let mut next_candidate = candidate.clone();
170            let moved_elements = 问题.操作.变异(&mut next_candidate);
171            let (_, next_energy) = 问题.计算(&next_candidate, &Some(moved_elements));
172            sum_delta += (next_energy - energy).abs();
173        }
174        let initial_guess = sum_delta / batch as f64;
175        let mut temperature = initial_guess;
176        let mut accept_rate;
177        let mut improve_rate;
178        (candidate, accept_rate, improve_rate) =
179            self.trial_run(问题, candidate, temperature, batch);
180        while accept_rate > HIGH_ACCEPTANCE {
181            temperature /= MULTIPLIER;
182            (candidate, accept_rate, improve_rate) =
183                self.trial_run(问题, candidate, temperature, batch);
184            界面.发送(消息::TrialMax {
185                temperature,
186                accept_rate,
187            });
188        }
189        while accept_rate < HIGH_ACCEPTANCE {
190            temperature *= MULTIPLIER;
191            (candidate, accept_rate, improve_rate) =
192                self.trial_run(问题, candidate, temperature, batch);
193            界面.发送(消息::TrialMax {
194                temperature,
195                accept_rate,
196            });
197        }
198        let t_max = temperature;
199        candidate = 问题.数据.初始映射.clone();
200        temperature = initial_guess;
201        while improve_rate > LOW_IMPROVEMENT {
202            temperature /= MULTIPLIER;
203            (candidate, _, improve_rate) = self.trial_run(问题, candidate, temperature, batch);
204            界面.发送(消息::TrialMin {
205                temperature,
206                improve_rate,
207            });
208        }
209        let t_min = temperature;
210        界面.发送(消息::Parameters { t_max, t_min });
211        降温时间表 {
212            t_max,
213            t_min,
214            steps: 1000,
215        }
216    }
