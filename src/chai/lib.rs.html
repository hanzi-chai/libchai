147/// 用于在图形界面验证输入的配置是否正确
148#[wasm_bindgen]
149pub fn validate(js_config: JsValue) -> Result<JsValue, JsError> {
150    set_once();
151    let 配置: 配置 = from_value(js_config)?;
152    let 序列化 = Serializer::json_compatible();
153    Ok(配置.serialize(&序列化)?)
154}

156#[wasm_bindgen]
157impl Web {
158    pub fn new(回调: Function) -> Web {
159        set_once();
160        let 参数 = 图形界面参数::default();
161        Self { 回调, 参数 }
162    }
163
164    pub fn sync(&mut self, 前端参数: JsValue) -> Result<(), JsError> {
165        self.参数 = from_value(前端参数)?;
166        Ok(())
167    }
168
169    pub fn encode_evaluate(&self, 前端目标函数配置: JsValue) -> Result<JsValue, JsError> {
170        let 目标函数配置: ObjectiveConfig = from_value(前端目标函数配置)?;
171        let 图形界面参数 {
172            mut 配置,
173            原始键位分布信息,
174            原始当量信息,
175            词列表,
176        } = self.参数.clone();
177        配置.optimization = Some(OptimizationConfig {
178            objective: 目标函数配置,
179            constraints: None,
180            metaheuristic: None,
181        });
182        let 数据 = 数据::新建(配置, 词列表, 原始键位分布信息, 原始当量信息)?;
183        let mut 编码器 = 默认编码器::新建(&数据)?;
184        let mut 编码结果 = 编码器.编码(&数据.初始映射, &None).clone();
185        let 码表 = 数据.生成码表(&编码结果);
186        let mut 目标函数 = 默认目标函数::新建(&数据)?;
187        let (指标, _) = 目标函数.计算(&mut 编码结果, &数据.初始映射);
188        Ok(to_value(&(码表, 指标))?)
189    }
190
191    pub fn optimize(&self) -> Result<(), JsError> {
192        let 图形界面参数 {
193            配置,
194            原始键位分布信息,
195            原始当量信息,
196            词列表,
197        } = self.参数.clone();
198        let 优化方法配置 = 配置.clone().optimization.unwrap().metaheuristic.unwrap();
199        let 数据: 数据 = 数据::新建(配置, 词列表, 原始键位分布信息, 原始当量信息)?;
200        let 编码器 = 默认编码器::新建(&数据)?;
201        let 目标函数 = 默认目标函数::新建(&数据)?;
202        let 操作 = 默认操作::新建(&数据)?;
203        let mut 问题 = 优化问题::新建(数据, 编码器, 目标函数, 操作);
204        let SolverConfig::SimulatedAnnealing(退火) = 优化方法配置;
205        退火.优化(&mut 问题, self);
206        Ok(())
207    }
208}

277impl 命令行 {
278    pub fn 新建(args: 命令行参数, maybe_output_dir: Option<PathBuf>) -> Self {
279        let output_dir = maybe_output_dir.unwrap_or_else(|| {
280            let time = Local::now().format("%m-%d+%H_%M_%S").to_string();
281            PathBuf::from(format!("output-{}", time))
282        });
283        create_dir_all(output_dir.clone()).unwrap();
284        Self {
285            参数: args,
286            输出目录: output_dir,
287        }
288    }
289
290    pub fn 读取(name: &str) -> 数据 {
291        let config = format!("examples/{}.yaml", name);
292        let elements = format!("examples/{}.txt", name);
293        let 参数 = 命令行参数 {
294            command: 命令::Optimize {
295                data: 数据参数 {
296                    config: Some(PathBuf::from(config)),
297                    encodables: Some(PathBuf::from(elements)),
298                    key_distribution: None,
299                    pair_equivalence: None,
300                },
301                threads: 1,
302            },
303        };
304        let cli: 命令行 = 命令行::新建(参数, None);
305        cli.准备数据()
306    }
307
308    fn read<I, T>(path: PathBuf) -> T
309    where
310        I: for<'de> Deserialize<'de>,
311        T: FromIterator<I>,
312    {
313        let mut reader = ReaderBuilder::new()
314            .delimiter(b'\t')
315            .has_headers(false)
316            .flexible(true)
317            .from_path(path)
318            .unwrap();
319        reader.deserialize().map(|x| x.unwrap()).collect()
320    }
321
322    pub fn 准备数据(&self) -> 数据 {
323        let (config, encodables, key_distribution, pair_equivalence) = match &self.参数.command {
324            命令::Encode { data } | 命令::Optimize { data, .. } => (
325                data.config.clone(),
326                data.encodables.clone(),
327                data.key_distribution.clone(),
328                data.pair_equivalence.clone(),
329            ),
330            命令::Server { .. } => {
331                panic!("Server 命令不需要数据准备");
332            }
333        };
334
335        let config_path = config.unwrap_or(PathBuf::from("config.yaml"));
336        let config_content = read_to_string(&config_path)
337            .unwrap_or_else(|_| panic!("文件 {} 不存在", config_path.display()));
338        let config: 配置 = serde_yaml::from_str(&config_content).unwrap();
339        let elements_path = encodables.unwrap_or(PathBuf::from("elements.txt"));
340        let encodables: Vec<原始可编码对象> = Self::read(elements_path);
341
342        let assets_dir = Path::new("assets");
343        let keq_path = key_distribution.unwrap_or(assets_dir.join("key_distribution.txt"));
344        let key_distribution: 原始键位分布信息 = Self::read(keq_path);
345        let peq_path = pair_equivalence.unwrap_or(assets_dir.join("pair_equivalence.txt"));
346        let pair_equivalence: 原始当量信息 = Self::read(peq_path);
347        数据::新建(config, encodables, key_distribution, pair_equivalence).unwrap()
348    }

350    pub fn 输出编码结果(&self, entries: Vec<码表项>) {
351        let path = self.输出目录.join("编码.txt");
352        let mut writer = WriterBuilder::new()
353            .delimiter(b'\t')
354            .has_headers(false)
355            .from_path(&path)
356            .unwrap();
357        for 码表项 {
358            name,
359            full,
360            full_rank,
361            short,
362            short_rank,
363        } in entries
364        {
365            writer
366                .serialize((&name, &full, &full_rank, &short, &short_rank))
367                .unwrap();
368        }
369        writer.flush().unwrap();
370        println!("已完成编码，结果保存在 {} 中", path.clone().display());
371    }
372
373    pub fn 输出评测指标<M: Display + Serialize>(&self, metric: M) {
374        let path = self.输出目录.join("评测指标.yaml");
375        print!("{}", metric);
376        let metric_str = serde_yaml::to_string(&metric).unwrap();
377        write(&path, metric_str).unwrap();
378    }
379
380    pub fn 生成子命令行(&self, index: usize) -> 命令行 {
381        let child_dir = self.输出目录.join(format!("{}", index));
382        命令行::新建(self.参数.clone(), Some(child_dir))
383    }
384}

386impl 界面 for 命令行 {
387    fn 发送(&self, message: 消息) {
388        let is_multithreaded = match &self.参数.command {
389            命令::Optimize { threads, .. } => *threads != 1,
390            _ => false,
391        };
392
393        let mut writer: Box<dyn Write> = if is_multithreaded {
394            let log_path = self.输出目录.join("log.txt");
395            let file = OpenOptions::new()
396                .create(true) // 如果文件不存在，则创建
397                .append(true) // 追加写入，不覆盖原有内容
398                .open(log_path)
399                .expect("Failed to open file");
400            Box::new(file)
401        } else {
402            Box::new(std::io::stdout())
403        };
404        let result = match message {
405            消息::TrialMax {
406                temperature,
407                accept_rate,
408            } => writeln!(
409                &mut writer,
410                "若温度为 {:.2e}，接受率为 {:.2}%",
411                temperature,
412                accept_rate * 100.0
413            ),
414            消息::TrialMin {
415                temperature,
416                improve_rate,
417            } => writeln!(
418                &mut writer,
419                "若温度为 {:.2e}，改进率为 {:.2}%",
420                temperature,
421                improve_rate * 100.0
422            ),
423            消息::Parameters { t_max, t_min } => writeln!(
424                &mut writer,
425                "参数寻找完成，从最高温 {} 降到最低温 {}……",
426                t_max, t_min
427            ),
428            消息::Elapsed { time } => writeln!(&mut writer, "计算一次评测用时：{} μs", time),
429            消息::Progress {
430                steps,
431                temperature,
432                metric,
433            } => writeln!(
434                &mut writer,
435                "已执行 {} 步，当前温度为 {:.2e}，当前评测指标如下：\n{}",
436                steps, temperature, metric
437            ),
438            消息::BetterSolution {
439                metric,
440                config,
441                save,
442            } => {
443                let 时刻 = Local::now();
444                let 时间戳 = 时刻.format("%m-%d+%H_%M_%S_%3f").to_string();
445                let 配置路径 = self.输出目录.join(format!("{}.yaml", 时间戳));
446                let 指标路径 = self.输出目录.join(format!("{}.txt", 时间戳));
447                if save {
448                    let mut 配置 = config.clone();
449                    if let Some(info) = 配置.info.as_mut() {
450                        info.version = Some(时间戳.clone());
451                    }
452                    let 序列化配置 = serde_yaml::to_string(&配置).unwrap();
453                    write(指标路径, metric.clone()).unwrap();
454                    write(配置路径, 序列化配置).unwrap();
455                    writeln!(
456                        &mut writer,
457                        "方案文件保存于 {}.yaml 中，评测指标保存于 {}.metric.yaml 中",
458                        时间戳, 时间戳
459                    )
460                    .unwrap();
461                }
462                writeln!(
463                    &mut writer,
464                    "{} 系统搜索到了一个更好的方案，评测指标如下：\n{}",
465                    时刻.format("%H:%M:%S"),
466                    metric
467                )
468            }
469        };
470        result.unwrap()
471    }
472}
