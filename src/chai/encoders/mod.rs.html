11pub trait 编码器 {
12    fn 编码(
13        &mut self,
14        keymap: &元素映射,
15        moved_elements: &Option<Vec<元素>>,
16    ) -> &mut Vec<编码信息>;
17}

19#[derive(Clone)]
20pub struct 编码空间 {
21    pub 线性表: Vec<u8>,
22    pub 线性表长度: usize,
23    pub 哈希表: FxHashMap<编码, u8>,
24}
25
26impl 编码空间 {
27    #[inline(always)]
28    pub fn 添加(&mut self, 编码: u64) {
29        if 编码 < self.线性表长度 as u64 {
30            let 编码 = 编码 as usize;
31            self.线性表[编码] = self.线性表[编码].saturating_add(1);
32        } else {
33            self.哈希表
34                .entry(编码)
35                .and_modify(|x| *x = x.saturating_add(1))
36                .or_insert(1);
37        }
38    }
39
40    #[inline(always)]
41    pub fn 查找数量(&self, 编码: u64) -> u8 {
42        if 编码 < self.线性表长度 as u64 {
43            self.线性表[编码 as usize]
44        } else {
45            *self.哈希表.get(&编码).unwrap_or(&0)
46        }
47    }
48}

50#[derive(Debug)]
51pub struct 简码配置 {
52    pub prefix: usize,
53    pub select_keys: Vec<键>,
54}
55
56pub struct 编码配置 {
57    pub 进制: u64,
58    pub 乘数列表: Vec<u64>,
59    pub 最大码长: usize,
60    pub 自动上屏查找表: 自动上屏,
61    pub 选择键: Vec<键>,
62    pub 首选键: 键,
63    pub 简码配置列表: Option<[Vec<简码配置>; 最大词长]>,
64}
65
66impl 编码配置 {
67    pub fn new(数据: &数据) -> Result<Self, 错误> {
68        let 编码器配置 = &数据.配置.encoder;
69        let 最大码长 = 编码器配置.max_length;
70        if 最大码长 >= 8 {
71            return Err("目前暂不支持最大码长大于等于 8 的方案计算！".into());
72        }
73        let 自动上屏查找表 = 数据.预处理自动上屏()?;
74        let mut 简码配置列表 = None;
75        if let Some(configs) = &编码器配置.short_code {
76            简码配置列表 = Some(数据.预处理简码配置(configs.clone())?);
77        }
78        Ok(Self {
79            自动上屏查找表,
80            最大码长,
81            进制: 数据.进制,
82            乘数列表: (0..=最大码长).map(|x| 数据.进制.pow(x as u32)).collect(),
83            选择键: 数据.选择键.clone(),
84            首选键: 数据.选择键[0],
85            简码配置列表,
86        })
87    }
88
89    #[inline(always)]
90    pub fn 生成编码(
91        &self, 原始编码: u64, 原始编码候选位置: u8, 选择键乘数: u64
92    ) -> u64 {
93        // 如果位于首选，检查是否能自动上屏，不能则加上首选键
94        if 原始编码候选位置 == 0 {
95            if *self.自动上屏查找表.get(原始编码 as usize).unwrap_or(&true) {
96                return 原始编码;
97            } else {
98                return 原始编码 + self.首选键 * 选择键乘数;
99            }
100        }
101        // 如果不是首选，无论如何都加上选择键
102        let 选择键 = *self
103            .选择键
104            .get(原始编码候选位置 as usize)
105            .unwrap_or(&self.选择键[0]);
106        原始编码 + 选择键 * 选择键乘数
107    }
108}
