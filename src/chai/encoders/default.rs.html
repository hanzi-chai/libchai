8#[derive(Clone)]
9pub struct 编码空间 {
10    pub 线性表: Vec<u8>,
11    pub 线性表长度: usize,
12    pub 哈希表: FxHashMap<编码, u8>,
13}
14
15impl 编码空间 {
16    #[inline(always)]
17    pub fn 添加(&mut self, 编码: u64) {
18        if 编码 < self.线性表长度 as u64 {
19            let 编码 = 编码 as usize;
20            self.线性表[编码] = self.线性表[编码].saturating_add(1);
21        } else {
22            self.哈希表
23                .entry(编码)
24                .and_modify(|x| *x = x.saturating_add(1))
25                .or_insert(1);
26        }
27    }
28
29    #[inline(always)]
30    pub fn 查找数量(&self, 编码: u64) -> u8 {
31        if 编码 < self.线性表长度 as u64 {
32            self.线性表[编码 as usize]
33        } else {
34            *self.哈希表.get(&编码).unwrap_or(&0)
35        }
36    }
37}

45pub struct 编码配置 {
46    pub 进制: u64,
47    pub 乘数列表: Vec<u64>,
48    pub 最大码长: usize,
49    pub 自动上屏查找表: 自动上屏,
50    pub 选择键: Vec<键>,
51    pub 首选键: 键,
52    pub 简码配置列表: Option<[Vec<简码配置>; 最大词长]>,
53}
54
55impl 编码配置 {
56    pub fn new(上下文: &默认上下文) -> Result<Self, 错误> {
57        let 编码器配置 = &上下文.配置.encoder;
58        let 最大码长 = 编码器配置.max_length;
59        if 最大码长 >= 8 {
60            return Err("目前暂不支持最大码长大于等于 8 的方案计算！".into());
61        }
62        let 自动上屏查找表 = 上下文.预处理自动上屏()?;
63        let mut 简码配置列表 = None;
64        if let Some(configs) = &编码器配置.short_code {
65            简码配置列表 = Some(上下文.预处理简码配置(configs.clone())?);
66        }
67        Ok(Self {
68            自动上屏查找表,
69            最大码长,
70            进制: 上下文.棱镜.进制,
71            乘数列表: (0..=最大码长).map(|x| 上下文.棱镜.进制.pow(x as u32)).collect(),
72            选择键: 上下文.选择键.clone(),
73            首选键: 上下文.选择键[0],
74            简码配置列表,
75        })
76    }
77
78    #[inline(always)]
79    pub fn 生成编码(
80        &self, 原始编码: u64, 原始编码候选位置: u8, 选择键乘数: u64
81    ) -> u64 {
82        // 如果位于首选，检查是否能自动上屏，不能则加上首选键
83        if 原始编码候选位置 == 0 {
84            if *self.自动上屏查找表.get(原始编码 as usize).unwrap_or(&true) {
85                return 原始编码;
86            } else {
87                return 原始编码 + self.首选键 * 选择键乘数;
88            }
89        }
90        // 如果不是首选，无论如何都加上选择键
91        let 选择键 = *self
92            .选择键
93            .get(原始编码候选位置 as usize)
94            .unwrap_or(&self.选择键[0]);
95        原始编码 + 选择键 * 选择键乘数
96    }
97}

99pub struct 默认编码器 {
100    编码配置: 编码配置,
101    词信息: Vec<可编码对象>,
102    全码空间: 编码空间,
103    简码空间: 编码空间,
104    包含元素的词: Vec<Vec<usize>>,
105}
106
107impl 默认编码器 {
108    /// 提供配置表示、拆分表、词表和共用资源来创建一个编码引擎
109    /// 字需要提供拆分表
110    /// 词只需要提供词表，它对应的拆分序列从字推出
111    pub fn 新建(上下文: &默认上下文) -> Result<Self, 错误> {
112        let 编码器配置 = &上下文.配置.encoder;
113        let 最大码长 = 编码器配置.max_length;
114        if 最大码长 >= 8 {
115            return Err("目前暂不支持最大码长大于等于 8 的方案计算！".into());
116        }
117        let 词信息 = 上下文.词列表.clone();
118        let 线性表长度 = 上下文.棱镜.进制.pow(最大码长 as u32) as usize;
119        let 全码空间 = 编码空间 {
120            线性表: vec![u8::default(); 线性表长度],
121            线性表长度,
122            哈希表: FxHashMap::default(),
123        };
124        let 简码空间 = 全码空间.clone();
125        let mut 包含元素的词 = vec![];
126        for _ in 0..=上下文.棱镜.元素转数字.len() {
127            包含元素的词.push(vec![]);
128        }
129        for (词序号, 词) in 词信息.iter().enumerate() {
130            for 元素 in &词.元素序列 {
131                包含元素的词[*元素].push(词序号);
132            }
133        }
134        let 编码配置 = 编码配置::new(上下文)?;
135        Ok(Self {
136            编码配置,
137            词信息,
138            全码空间,
139            简码空间,
140            包含元素的词,
141        })
142    }
143
144    fn 重置(&mut self) {
145        self.全码空间.线性表.iter_mut().for_each(|x| {
146            *x = 0;
147        });
148        self.全码空间.哈希表.clear();
149        self.简码空间.线性表.iter_mut().for_each(|x| {
150            *x = 0;
151        });
152        self.简码空间.哈希表.clear();
153    }
154
155    fn 输出全码(
156        &mut self,
157        映射: &元素映射,
158        移动的元素: &Option<Vec<元素>>,
159        编码结果: &mut [编码信息],
160    ) {
161        let 编码配置 = &self.编码配置;
162        // 根据移动的元素更新编码结果，如果没有移动的元素则直接全部生成
163        if let Some(移动的元素) = 移动的元素 {
164            for 元素 in 移动的元素 {
165                for 索引 in &self.包含元素的词[*元素] {
166                    let 词 = &self.词信息[*索引];
167                    let 全码信息 = &mut 编码结果[*索引].全码;
168                    let mut 原始编码 = 0;
169                    for (元素, 乘数) in zip(&词.元素序列, &编码配置.乘数列表) {
170                        原始编码 += 映射[*元素] * 乘数;
171                    }
172                    全码信息.原始编码 = 原始编码;
173                }
174            }
175        } else {
176            for (词, 编码信息) in zip(&self.词信息, 编码结果.iter_mut()) {
177                let 全码信息 = &mut 编码信息.全码;
178                let mut 原始编码 = 0;
179                for (元素, 乘数) in zip(&词.元素序列, &编码配置.乘数列表) {
180                    原始编码 += 映射[*元素] * 乘数;
181                }
182                全码信息.原始编码 = 原始编码;
183            }
184        }
185
186        for (编码信息, 词) in 编码结果.iter_mut().zip(&self.词信息) {
187            let 全码信息 = &mut 编码信息.全码;
188            // 首先查找编码空间中已经有了多少个同样原始编码的词，确定候选位置
189            let 原始编码候选位置 = self.全码空间.查找数量(全码信息.原始编码);
190            全码信息.原始编码候选位置 = 原始编码候选位置;
191            self.全码空间.添加(全码信息.原始编码);
192            // 然后生成实际编码，并向全码信息中写入实际编码和实际编码是否重码的信息，用于测评
193            // 注意：对于全码来说，暂且忽略次选及之后的选择键的影响，统一视为首选进行编码。这可以避免在四码类方案中大量出现五码的编码，影响性能
194            let 乘数 = 编码配置.乘数列表[词.元素序列.len()];
195            let 编码 = 编码配置.生成编码(全码信息.原始编码, 0, 乘数);
196            let 是否重码 = 原始编码候选位置 > 0;
197            全码信息.更新(编码, 是否重码);
198        }
199    }
200
201    fn 输出简码(&mut self, 编码结果: &mut [编码信息]) {
202        let 编码配置 = &self.编码配置;
203        let 简码配置列表 = 编码配置.简码配置列表.as_ref().unwrap();
204        // 优先简码
205        for (词, 编码结果) in zip(&self.词信息, 编码结果.iter_mut()) {
206            if 词.简码等级 == u64::MAX {
207                continue;
208            }
209            let 原始编码 = 编码结果.全码.原始编码 % 编码配置.乘数列表[词.简码等级 as usize];
210            编码结果.简码.原始编码 = 原始编码;
211            let 序号 = self.简码空间.查找数量(原始编码);
212            let 乘数 = 编码配置.乘数列表[词.简码等级 as usize];
213            let 编码 = 编码配置.生成编码(原始编码, 序号, 乘数);
214            编码结果.简码.更新(编码, 序号 > 0);
215            self.简码空间.添加(原始编码);
216        }
217        // 常规简码
218        for (词, 编码结果) in zip(&self.词信息, 编码结果.iter_mut()) {
219            if 词.简码等级 != u64::MAX {
220                continue;
221            }
222            let 简码配置 = &简码配置列表[词.词长 - 1];
223            let mut 有简码 = false;
224            let 全码信息 = &编码结果.全码;
225            let 简码信息 = &mut 编码结果.简码;
226            for 出简方式 in 简码配置 {
227                let 简码配置 {
228                    prefix,
229                    select_keys,
230                } = 出简方式;
231                let 乘数 = 编码配置.乘数列表[*prefix];
232                // 如果根本没有这么多码，就放弃
233                if 全码信息.原始编码 < 乘数 {
234                    continue;
235                }
236                // 将全码截取一部分出来，检查当前简码位置上的候选数量是否达到上限
237                let 原始编码 = 全码信息.原始编码 % 乘数;
238                let 序号 = self.全码空间.查找数量(原始编码) + self.简码空间.查找数量(原始编码);
239                if 序号 >= select_keys.len() as u8 {
240                    continue;
241                }
242                // 如果没有达到上限，就可以出这个简码
243                let 编码 = 编码配置.生成编码(原始编码, 序号, 乘数);
244                简码信息.原始编码 = 原始编码;
245                简码信息.原始编码候选位置 = 序号;
246                简码信息.更新(编码, false);
247                self.简码空间.添加(原始编码);
248                有简码 = true;
249                break;
250            }
251            if !有简码 {
252                let 序号 = self.简码空间.查找数量(全码信息.原始编码);
253                简码信息.原始编码 = 全码信息.原始编码;
254                简码信息.原始编码候选位置 = 序号;
255                简码信息.更新(全码信息.实际编码, 序号 > 0);
256                self.简码空间.添加(全码信息.原始编码);
257            }
258        }
259    }
260}

262impl 编码器 for 默认编码器 {
263    type 解类型 = 元素映射;
264    fn 编码(
265        &mut self,
266        映射: &元素映射,
267        移动的元素: &Option<Vec<元素>>,
268        输出: &mut [编码信息],
269    ) {
270        self.重置();
271        self.输出全码(映射, 移动的元素, 输出);
272        if self.编码配置.简码配置列表.is_none()
273            || self.编码配置.简码配置列表.as_ref().unwrap().is_empty()
274        {
275            return;
276        }
277        self.输出简码(输出);
278    }
