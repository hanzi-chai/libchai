7pub struct 默认编码器 {
8    编码结果: Vec<编码信息>,
9    编码配置: 编码配置,
10    词信息: Vec<可编码对象>,
11    全码空间: 编码空间,
12    简码空间: 编码空间,
13    包含元素的词: Vec<Vec<usize>>,
14}
15
16impl 默认编码器 {
17    /// 提供配置表示、拆分表、词表和共用资源来创建一个编码引擎
18    /// 字需要提供拆分表
19    /// 词只需要提供词表，它对应的拆分序列从字推出
20    pub fn 新建(数据: &数据) -> Result<Self, 错误> {
21        let 编码器配置 = &数据.配置.encoder;
22        let 最大码长 = 编码器配置.max_length;
23        if 最大码长 >= 8 {
24            return Err("目前暂不支持最大码长大于等于 8 的方案计算！".into());
25        }
26        let 词信息 = 数据.词列表.clone();
27        let 编码结果 = 词信息.iter().map(编码信息::new).collect();
28        let 线性表长度 = 数据.进制.pow(最大码长 as u32) as usize;
29        let 全码空间 = 编码空间 {
30            线性表: vec![u8::default(); 线性表长度],
31            线性表长度,
32            哈希表: FxHashMap::default(),
33        };
34        let 简码空间 = 全码空间.clone();
35        let mut 包含元素的词 = vec![];
36        for _ in 0..=数据.元素转数字.len() {
37            包含元素的词.push(vec![]);
38        }
39        for (词序号, 词) in 词信息.iter().enumerate() {
40            for 元素 in &词.元素序列 {
41                包含元素的词[*元素].push(词序号);
42            }
43        }
44        let 编码配置 = 编码配置::new(数据)?;
45        Ok(Self {
46            编码结果,
47            编码配置,
48            词信息,
49            全码空间,
50            简码空间,
51            包含元素的词,
52        })
53    }
54
55    fn 重置(&mut self) {
56        self.全码空间.线性表.iter_mut().for_each(|x| {
57            *x = 0;
58        });
59        self.全码空间.哈希表.clear();
60        self.简码空间.线性表.iter_mut().for_each(|x| {
61            *x = 0;
62        });
63        self.简码空间.哈希表.clear();
64    }
65
66    fn 输出全码(&mut self, 映射: &元素映射, 移动的元素: &Option<Vec<元素>>) {
67        let 编码配置 = &self.编码配置;
68        let 编码结果 = &mut self.编码结果;
69        // 根据移动的元素更新编码结果，如果没有移动的元素则直接全部生成
70        if let Some(移动的元素) = 移动的元素 {
71            for 元素 in 移动的元素 {
72                for 索引 in &self.包含元素的词[*元素] {
73                    let 词 = &self.词信息[*索引];
74                    let 全码信息 = &mut 编码结果[*索引].全码;
75                    let mut 原始编码 = 0;
76                    for (元素, 乘数) in zip(&词.元素序列, &编码配置.乘数列表) {
77                        原始编码 += 映射[*元素] * 乘数;
78                    }
79                    全码信息.原始编码 = 原始编码;
80                }
81            }
82        } else {
83            for (词, 编码信息) in zip(&self.词信息, 编码结果.iter_mut()) {
84                let 全码信息 = &mut 编码信息.全码;
85                let mut 原始编码 = 0;
86                for (元素, 乘数) in zip(&词.元素序列, &编码配置.乘数列表) {
87                    原始编码 += 映射[*元素] * 乘数;
88                }
89                全码信息.原始编码 = 原始编码;
90            }
91        }
92
93        for (编码信息, 词) in 编码结果.iter_mut().zip(&self.词信息) {
94            let 全码信息 = &mut 编码信息.全码;
95            // 首先查找编码空间中已经有了多少个同样原始编码的词，确定候选位置
96            let 原始编码候选位置 = self.全码空间.查找数量(全码信息.原始编码);
97            全码信息.原始编码候选位置 = 原始编码候选位置;
98            self.全码空间.添加(全码信息.原始编码);
99            // 然后生成实际编码，并向全码信息中写入实际编码和实际编码是否重码的信息，用于测评
100            // 注意：对于全码来说，暂且忽略次选及之后的选择键的影响，统一视为首选进行编码。这可以避免在四码类方案中大量出现五码的编码，影响性能
101            let 乘数 = 编码配置.乘数列表[词.元素序列.len()];
102            let 编码 = 编码配置.生成编码(全码信息.原始编码, 0, 乘数);
103            let 是否重码 = 原始编码候选位置 > 0;
104            全码信息.更新(编码, 是否重码);
105        }
106    }
107
108    fn 输出简码(&mut self) {
109        let 编码配置 = &self.编码配置;
110        let 编码结果 = &mut self.编码结果;
111        let 简码配置列表 = 编码配置.简码配置列表.as_ref().unwrap();
112        // 优先简码
113        for (词, 编码结果) in zip(&self.词信息, 编码结果.iter_mut()) {
114            if 词.简码等级 == u64::MAX {
115                continue;
116            }
117            let 原始编码 = 编码结果.全码.原始编码 % 编码配置.乘数列表[词.简码等级 as usize];
118            编码结果.简码.原始编码 = 原始编码;
119            let 序号 = self.简码空间.查找数量(原始编码);
120            let 乘数 = 编码配置.乘数列表[词.简码等级 as usize];
121            let 编码 = 编码配置.生成编码(原始编码, 序号, 乘数);
122            编码结果.简码.更新(编码, 序号 > 0);
123            self.简码空间.添加(原始编码);
124        }
125        // 常规简码
126        for (词, 编码结果) in zip(&self.词信息, 编码结果.iter_mut()) {
127            if 词.简码等级 != u64::MAX {
128                continue;
129            }
130            let 简码配置 = &简码配置列表[词.词长 - 1];
131            let mut 有简码 = false;
132            let 全码信息 = &编码结果.全码;
133            let 简码信息 = &mut 编码结果.简码;
134            for 出简方式 in 简码配置 {
135                let 简码配置 {
136                    prefix,
137                    select_keys,
138                } = 出简方式;
139                let 乘数 = 编码配置.乘数列表[*prefix];
140                // 如果根本没有这么多码，就放弃
141                if 全码信息.原始编码 < 乘数 {
142                    continue;
143                }
144                // 将全码截取一部分出来，检查当前简码位置上的候选数量是否达到上限
145                let 原始编码 = 全码信息.原始编码 % 乘数;
146                let 序号 = self.全码空间.查找数量(原始编码) + self.简码空间.查找数量(原始编码);
147                if 序号 >= select_keys.len() as u8 {
148                    continue;
149                }
150                // 如果没有达到上限，就可以出这个简码
151                let 编码 = 编码配置.生成编码(原始编码, 序号, 乘数);
152                简码信息.原始编码 = 原始编码;
153                简码信息.原始编码候选位置 = 序号;
154                简码信息.更新(编码, false);
155                self.简码空间.添加(原始编码);
156                有简码 = true;
157                break;
158            }
159            if !有简码 {
160                let 序号 = self.简码空间.查找数量(全码信息.原始编码);
161                简码信息.原始编码 = 全码信息.原始编码;
162                简码信息.原始编码候选位置 = 序号;
163                简码信息.更新(全码信息.实际编码, 序号 > 0);
164                self.简码空间.添加(全码信息.原始编码);
165            }
166        }
167    }
168}

170impl 编码器 for 默认编码器 {
171    fn 编码(
172        &mut self,
173        映射: &元素映射,
174        移动的元素: &Option<Vec<元素>>,
175    ) -> &mut Vec<编码信息> {
176        self.重置();
177        self.输出全码(映射, 移动的元素);
178        if self.编码配置.简码配置列表.is_none()
179            || self.编码配置.简码配置列表.as_ref().unwrap().is_empty()
180        {
181            return &mut self.编码结果;
182        }
183        self.输出简码();
184        &mut self.编码结果
185    }
